```python
# Every phase module exposes this signature:
def run(
    episode_id: str,
    asset_bus_read: Callable[[str], Any],
    asset_bus_write: Callable[[str, dict], None],
    delegate_task: Callable[[str, str, list[str]], dict],
    trigger_gate: Callable[[str, str], dict] | None = None,
) -> dict:
    """Returns {"phase": PHASE_ID, "outputs": {...}, "gate": {...} | None}"""

# Module-level constants:
PHASE_ID: str            # e.g. "p01_hook_topic"
EXPERT: str              # primary expert name (for documentation)
INPUT_SLOTS: list[str]   # asset-bus slots read
OUTPUT_SLOTS: list[str]  # asset-bus slots written
GATE_ID: str | None      # gate_id from references/review-gates.md, or None if no gate

# delegate_task returns {"summary": "<text with fenced JSON block>", ...}
# Phase module extracts structured output via _parse_expert_output(result)
# which finds the last ```json ... ``` block and json.loads it.

# Gate invocation contract:
# trigger_gate(gate_id, episode_id) -> dict
# Phase 34's runner_hooks.pause_for_review(gate_id, episode_id, payload, mode=None)
# is the production wiring (set up by runner.py from 35-02).
```
