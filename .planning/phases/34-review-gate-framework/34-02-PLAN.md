### `gate_config.py` (~80-120 LOC)

```python
"""gate_config.py — gates.yaml loader + validator (Phase 34-02, SC#2).

Loads gates.yaml once at import time. Exposes GATE_REGISTRY: dict[str, dict]
(raw YAML) and to_gate_config(gate_id) -> GateConfig (frozen dataclass from
Plan 34-01).

D-34-02: hot-reload NOT supported — restart required to pick up YAML changes.
D-34-03: pyyaml is the ONE allowed third-party dep (already transitive via
plugin loader).
"""
from __future__ import annotations

import logging
from pathlib import Path
from typing import Any

import yaml  # pyyaml — already transitive dep (plugin loader uses it for plugin.yaml)

from plugins.review_gates.gate import GateConfig, GateMode

logger = logging.getLogger(__name__)

class GateConfigError(Exception):
    """Raised when gates.yaml is missing, malformed, or fails validation."""


_YAML_PATH = Path(__file__).parent / "gates.yaml"


def load_gates() -> dict[str, dict]:
    """Load and validate gates.yaml. Returns {gate_id: raw_yaml_dict}.

    Raises GateConfigError on missing file, malformed YAML, or failed validation.
    Idempotent (re-reads file each call; no caching at this layer — callers
    cache the result).
    """
    if not _YAML_PATH.exists():
        raise GateConfigError(f"gates.yaml not found at {_YAML_PATH}")
    try:
        with open(_YAML_PATH, encoding="utf-8") as f:
            doc = yaml.safe_load(f)
    except yaml.YAMLError as exc:
        raise GateConfigError(f"gates.yaml parse error: {exc}") from exc

    if not isinstance(doc, dict) or "gates" not in doc:
        raise GateConfigError("gates.yaml missing top-level 'gates' list")
    gate_list = doc["gates"]
    if not isinstance(gate_list, list) or len(gate_list) != 8:
        raise GateConfigError(
            f"gates.yaml must contain exactly 8 gates (found {len(gate_list) if isinstance(gate_list, list) else 'non-list'})"
        )

    registry: dict[str, dict] = {}
    for entry in gate_list:
        _validate_entry(entry)
        registry[entry["gate_id"]] = entry
    return registry


REQUIRED_FIELDS = (
    "gate_id", "phase", "asset_bus_slots_to_lock",
    "reviewer_role", "timeout_sec", "default_mode", "retry_policy",
)


def _validate_entry(entry: dict) -> None:
    if not isinstance(entry, dict):
        raise GateConfigError(f"gate entry is not a dict: {entry!r}")
    for field in REQUIRED_FIELDS:
        if field not in entry:
            raise GateConfigError(
                f"gate '{entry.get('gate_id', '<unknown>')}' missing required field '{field}'"
            )
    if not isinstance(entry["asset_bus_slots_to_lock"], list) or not entry["asset_bus_slots_to_lock"]:
        raise GateConfigError(
            f"gate '{entry['gate_id']}': asset_bus_slots_to_lock must be a non-empty list"
        )
    if not isinstance(entry["timeout_sec"], int) or entry["timeout_sec"] <= 0:
        raise GateConfigError(
            f"gate '{entry['gate_id']}': timeout_sec must be positive int"
        )
    if entry["default_mode"] not in {"blocking", "webhook", "polling"}:
        raise GateConfigError(
            f"gate '{entry['gate_id']}': default_mode must be blocking/webhook/polling"
        )
    rp = entry["retry_policy"]
    if not isinstance(rp, dict) or "max_retries" not in rp or "backoff_sec" not in rp:
        raise GateConfigError(
            f"gate '{entry['gate_id']}': retry_policy must have max_retries + backoff_sec"
        )
    if not isinstance(rp["max_retries"], int) or rp["max_retries"] <= 0:
        raise GateConfigError(
            f"gate '{entry['gate_id']}': retry_policy.max_retries must be positive int"
        )


def to_gate_config(gate_id: str) -> GateConfig:
    """Convert a raw YAML entry to a frozen GateConfig dataclass."""
    registry = load_gates()
    if gate_id not in registry:
        raise GateConfigError(f"Unknown gate_id: {gate_id}")
    entry = registry[gate_id]
    # Normalize reviewer_role list → comma-joined string
    rr = entry["reviewer_role"]
    if isinstance(rr, list):
        rr = ", ".join(rr)
    rp = entry["retry_policy"]
    return GateConfig(
        gate_id=entry["gate_id"],
        phase=entry["phase"],
        asset_bus_slots_to_lock=tuple(entry["asset_bus_slots_to_lock"]),
        reviewer_role=rr,
        timeout_sec=entry["timeout_sec"],
        callback_url=entry.get("callback_url"),
        max_retries=rp["max_retries"],
        backoff_sec=rp["backoff_sec"],
        default_mode=GateMode(entry["default_mode"]),
    )


# Eager load on import — fails loud if YAML is broken (D-34-02)
GATE_REGISTRY: dict[str, dict] = load_gates()
```
