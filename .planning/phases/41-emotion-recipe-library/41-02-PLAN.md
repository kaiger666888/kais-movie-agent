Wilson CI formula (CONTEXT.md — pure stdlib, z=1.96 for 95% CI):
```python
import math

def _wilson_ci(passed: int, total: int, z: float = 1.96) -> tuple[float, float]:
    """Return (lower, upper) Wilson score interval bounds at 95% CI."""
    if total == 0:
        return (0.0, 1.0)   # widest possible interval when no data
    p = passed / total
    denom = 1 + z * z / total
    center = (p + z * z / (2 * total)) / denom
    spread = z * math.sqrt((p * (1 - p) + z * z / (4 * total)) / total) / denom
    return (center - spread, center + spread)
```

```python
def update_validation(self, recipe_id: str, platform: str,
                     completion_rate: float, sample_size_delta: int = 1) -> dict:
    # 1. Read latest version of recipe (use get_recipe from 41-01)
    latest = self.get_recipe(recipe_id)   # raises KeyError if unknown

    # 2. Compute new validation{} fields:
    old_v = latest["validation"]
    new_sample_size = old_v["sample_size"] + sample_size_delta
    # running average: blend old completion_rate (weighted by old sample_size) with new
    if new_sample_size > 0:
        cumulative_passed = old_v["completion_rate"] * old_v["sample_size"]
        new_completion_rate = (cumulative_passed + completion_rate * sample_size_delta) / new_sample_size
    else:
        new_completion_rate = 0.0
    lower, upper = _wilson_ci(int(round(new_completion_rate * new_sample_size)), new_sample_size)
    ci_str = f"±{int(round((upper - lower) / 2 * 100))}%"
    new_converged = (new_sample_size >= 10) and ((upper - lower) <= 0.10)

    # 3. Append new row with version = latest+1, last_validated = now ISO 8601
    new_row = copy.deepcopy(latest)
    new_row["version"] = latest["version"] + 1
    new_row["validation"] = {
        "platform": platform,
        "completion_rate": new_completion_rate,
        "confidence_interval": ci_str,
        "sample_size": new_sample_size,
        "converged": new_converged,
    }
    new_row["provenance"]["last_validated"] = datetime.now(timezone.utc).isoformat()

    # 4. Append + return (NEVER mutate latest)
    try:
        self._bus.append_line("emotion-recipe", new_row)
        return new_row
    except Exception as e:
        logger.warning("RecipeLibrary update_validation degraded: %s", e)
        return None   # OR raise — action specifies; recommend return None to mirror 41-01
```

    ```python
    def _wilson_ci(passed: int, total: int, z: float = 1.96) -> tuple[float, float]:
        """Wilson score confidence interval (pure stdlib — uses math.sqrt).

        Returns (lower, upper) bounds at the given z-score (default 1.96 = 95% CI).
        Returns (0.0, 1.0) — the widest possible interval — when total == 0.

        Formula (CONTEXT.md "Wilson confidence interval"):
            p_hat = passed / total
            denom = 1 + z^2 / total
            center = (p_hat + z^2 / (2 * total)) / denom
            spread = z * sqrt((p_hat * (1 - p_hat) + z^2 / (4 * total)) / total) / denom
        """
        if total <= 0:
            return (0.0, 1.0)
        p = passed / total
        denom = 1 + z * z / total
        center = (p + z * z / (2 * total)) / denom
        spread = z * math.sqrt((p * (1 - p) + z * z / (4 * total)) / total) / denom
        return (center - spread, center + spread)
    ```
