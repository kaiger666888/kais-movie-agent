def _slugify(genre: str) -> str:
    # lowercase → replace whitespace with '-' → strip non-alphanumeric (keep '-') → collapse consecutive '-'
    s = genre.lower().strip()
    s = re.sub(r"\s+", "-", s)
    s = re.sub(r"[^a-z0-9-]", "", s)         # Chinese chars become '' (acceptable — slug is ASCII)
    s = re.sub(r"-+", "-", s)
    return s.strip("-")

def _next_sequence(bus: AssetBus, slug: str) -> int:
    # read all emotion-recipe rows; filter by recipe_id startswith f"{slug}-"; max+1
    rows = bus.read_lines("emotion-recipe")
    prefix = f"{slug}-"
    nums = [
        int(r["recipe_id"][len(prefix):])
        for r in rows
        if r.get("recipe_id", "").startswith(prefix)
        and r["recipe_id"][len(prefix):].isdigit()
    ]
    return (max(nums) + 1) if nums else 1
