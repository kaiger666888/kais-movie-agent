```python
from __future__ import annotations
import logging
from typing import Any, Callable, Optional
import httpx
from plugins.kais_aigc.canvas import CanvasClient
from plugins.kais_aigc import canvas_graph

logger = logging.getLogger(__name__)


class CanvasSyncSubscriber:
    """Python port of Node.js lib/canvas-sync-hook.js.

    Listens for phase completion + gate resolution events, builds FlowGraph
    upserts via canvas_graph pure functions, persists via CanvasClient HTTP.
    Degrade-tolerant — never raises to the runner (CANVAS-IN-HERMES-03).
    """

    def __init__(self, canvas: CanvasClient, agent_name: str = ""):
        self._client = canvas
        self._agent_name = agent_name
        self._prev_phase_id: Optional[str] = None

    def on_phase_complete(self, episode_id: str, phase_id: str, result: dict) -> None:
        """Phase completion handler. Builds node from result, upserts into
        graph, draws link from prev phase, saves via :10588 save-v2."""
        try:
            loaded = self._client.load_canvas()
            graph = canvas_graph.normalize_loaded_graph(
                loaded, self._client._project_id, self._client._episodes_id
            )
            node_data = self._build_phase_node(phase_id, result)
            canvas_graph.upsert_node(graph, f"n-{phase_id}", node_data)
            if self._prev_phase_id is not None:
                canvas_graph.ensure_link(
                    graph,
                    f"l-{self._prev_phase_id}-{phase_id}",
                    f"n-{self._prev_phase_id}",
                    f"n-{phase_id}",
                )
            self._client.save_canvas(graph)
            self._prev_phase_id = phase_id
        except Exception:
            logger.warning(
                "canvas sync on_phase_complete degraded (episode=%s phase=%s)",
                episode_id, phase_id, exc_info=True,
            )

    def on_gate_resolved(self, episode_id: str, gate_id: str, decision: str, payload: dict) -> None:
        """Gate resolution handler. On approve — write/update formal canvas
        node with reviewStatus=approved. On reject — mark phase node error."""
        try:
            loaded = self._client.load_canvas()
            graph = canvas_graph.normalize_loaded_graph(
                loaded, self._client._project_id, self._client._episodes_id
            )
            if decision == "approve":
                node_data = self._build_gate_node(gate_id, payload, approved=True)
                canvas_graph.upsert_node(graph, f"g-{gate_id}", node_data)
            else:  # reject / contest
                # mark the phase node associated with the gate as error
                phase_id = payload.get("phase_id") or gate_id
                canvas_graph.upsert_node(
                    graph, f"n-{phase_id}",
                    {"state": "error", "data": {"state": "error", "tags": ["rejected"]}},
                )
            self._client.save_canvas(graph)
        except Exception:
            logger.warning(
                "canvas sync on_gate_resolved degraded (gate=%s decision=%s)",
                gate_id, decision, exc_info=True,
            )

def register_canvas_sync(
    *,
    base_url: Optional[str],
    project_id: int,
    episodes_id: int,
    runner_config,  # RunnerConfig type — avoid circular import, use duck typing
    transport: Optional[httpx.BaseTransport] = None,
```
