```python
def kais_canvas_sync(
    project_id: int,
    episodes_id: int,
    base_url: Optional[str] = None,
    runner_config: Any = None,  # RunnerConfig — duck-typed to avoid circular import
) -> dict:
    """Register canvas sync subscriber on the supplied RunnerConfig.

    Wires both trigger paths (phase completion + gate resolution approve)
    in a single call. Returns the subscriber handle (caller must retain
    reference to prevent GC of the callback target).

    Args:
        project_id: canvas project ID
        episodes_id: canvas episodes ID
        base_url: canvas platform URL (defaults to KAIS_CANVAS_URL env or
            CanvasClient.DEFAULT_BASE_URL)
        runner_config: the RunnerConfig instance to wire callbacks onto

    Returns:
        {"registered": True, "subscriber": <CanvasSyncSubscriber>} or
        degrade envelope on construction failure.
    """
    from plugins.kais_aigc.canvas_sync import register_canvas_sync
    try:
        sub = register_canvas_sync(
            base_url=base_url, project_id=project_id,
            episodes_id=episodes_id, runner_config=runner_config,
        )
        return {"registered": True, "subscriber": sub}
    except Exception as exc:
        logger.warning("kais_canvas_sync registration failed: %s", exc, exc_info=True)
        return {"registered": False, "reason": str(exc)}
```
